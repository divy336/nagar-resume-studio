 // counters (1 default entry already exists)
let eduCount = 1;
let expCount = 1;
let projCount = 1;

// max limits
const MAX_EDU = 2;
const MAX_EXP = 2;
const MAX_PROJ = 3;


// 🔥 REMOVE BUTTON
function createRemoveBtn(el, type) {
    const btn = document.createElement('button');
    btn.className = 'remove-btn';
    btn.innerText = 'Remove';

    btn.onclick = () => {
        el.remove();

        if (type === "edu") eduCount--;
        if (type === "exp") expCount--;
        if (type === "proj") projCount--;

        updateButtons();
    };

    el.appendChild(btn);
}


// 🔥 UPDATE BUTTON + MESSAGE
function updateButtons() {

    const eduBtn = document.querySelector('[onclick="addEducation()"]');
    const expBtn = document.querySelector('[onclick="addExperience()"]');
    const projBtn = document.querySelector('[onclick="addProject()"]');

    const eduMsg = document.getElementById("eduMsg");
    const expMsg = document.getElementById("expMsg");
    const projMsg = document.getElementById("projMsg");

    // EDUCATION
    if (eduCount >= MAX_EDU) {
        eduBtn.disabled = true;
        if (eduMsg) eduMsg.innerText = "Max 2 education allowed";
    } else {
        eduBtn.disabled = false;
        if (eduMsg) eduMsg.innerText = "";
    }

    // EXPERIENCE
    if (expCount >= MAX_EXP) {
        expBtn.disabled = true;
        if (expMsg) expMsg.innerText = "Max 2 internships allowed";
    } else {
        expBtn.disabled = false;
        if (expMsg) expMsg.innerText = "";
    }

    // PROJECT
    if (projCount >= MAX_PROJ) {
        projBtn.disabled = true;
        if (projMsg) projMsg.innerText = "Max 3 projects allowed";
    } else {
        projBtn.disabled = false;
        if (projMsg) projMsg.innerText = "";
    }
}


// 🔥 EDUCATION
function addEducation() {
    if (eduCount >= MAX_EDU) return;

    eduCount++;

    const container = document.getElementById('educationContainer');
    const div = document.createElement('div');
    div.className = 'entry-card';

    div.innerHTML = `
        <div class="grid">
            <div class="form-group"><input name="degree[]" placeholder="Degree" value="{{ resume.get('degree', '') }}"></div>
            <div class="form-group"><input name="college[]" placeholder="Institute" value="{{ resume.get('college', '') }}"></div>
            <div class="form-group"><input name="board[]" placeholder="Board" value="{{ resume.get('board', '') }}"></div>
            <div class="form-group"><input name="cgpa[]" placeholder="CGPA" value="{{ resume.get('cgpa', '') }}"></div>
            <div class="form-group"><input name="edu_start[]" placeholder="Start Year" value="{{ resume.get('edu_start', '') }}"></div>
            <div class="form-group"><input name="edu_end[]" placeholder="End Year" value="{{ resume.get('edu_end', '') }}">
        </div>`;

    createRemoveBtn(div, "edu");
    container.appendChild(div);

    updateButtons();
}


// 🔥 EXPERIENCE
function addExperience() {
    if (expCount >= MAX_EXP) return;

    expCount++;

    const container = document.getElementById('experienceContainer');
    const div = document.createElement('div');
    div.className = 'entry-card';

    div.innerHTML = `
        <div class="grid">
            <div class="form-group"><input name="exp_role[]" placeholder="Role" value="{{ resume.get('exp_role', '') }}"></div>
            <div class="form-group"><input name="exp_company[]" placeholder="Company" value="{{ resume.get('exp_company', '') }}"></div>
        </div>
        <div class="form-group"><input name="exp_duration[]" placeholder="Duration" value="{{ resume.get('exp_duration', '') }}"></div>
        <div class="form-group"><textarea name="exp_desc[]" placeholder="Description">{{ resume.get('exp_desc', '') }}</textarea></div>`;
    createRemoveBtn(div, "exp");
    container.appendChild(div);

    updateButtons();
}


// 🔥 PROJECT
function addProject() {
    if (projCount >= MAX_PROJ) return;

    projCount++;

    const container = document.getElementById('projectContainer');
    const div = document.createElement('div');
    div.className = 'entry-card';

    div.innerHTML = `
        <div class="form-group"><input name="proj_title[]" placeholder="Project Title" value="{{ resume.get('proj_title', '') }}"></div>
        <div class="form-group"><input name="proj_tech[]" placeholder="Technologies"></div>
        <div class="form-group"><textarea name="proj_desc[]" placeholder="Description"></textarea></div>`;

    createRemoveBtn(div, "proj");
    container.appendChild(div);

    updateButtons();
}


// 🔥 INITIAL CALL
updateButtons();
